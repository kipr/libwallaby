495  {
496 #ifdef _WIN32
497  FILETIME ft;
498  GetSystemTimeAsFileTime(&ft);
499  unsigned long t = (ULONGLONG)ft.dwLowDateTime + ((ULONGLONG)(ft.dwHighDateTime) << 32);
500 
501  timeval tv;

503  tv.tv_sec = (t - tv.tv_usec) / 1000L;
